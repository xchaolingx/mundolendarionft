function fetchNFTMetadata(walletAddress, nftId) { // Obter acesso ao contrato da NFT // (neste exemplo, supomos que existe uma variável "nftContract" que contém a referência ao contrato) const contract = nftContract; // Buscar os metadados da NFT no contrato const metadata = contract.methods.getMetadata(nftId).call({ from: walletAddress }); // Retornar os metadados return metadata; }
