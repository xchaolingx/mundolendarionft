function connectToWallet() { if (typeof window.ethereum !== 'undefined') { window.web3 = new Web3(ethereum); try { ethereum.enable(); } catch (error) { console.error(error); } } else if (typeof window.web3 !== 'undefined') { window.web3 = new Web3(web3.currentProvider); } else { console.error('Nenhum provedor de Web3 encontrado'); } }
